#!/usr/bin/env bash
# Usage: lfphp-get gcc <VERSION> [--compile]
# Example: "lfphp-get gcc gcc-9.3.0 --compile"
export COMPILE=0
export response="gcc-9.3.0"
export flags=""
# this needs to change!
export source="http://opensource.unlikelysource.com"
if [[ -z "$1" ]]; then
    read -r -p "Which version of GCC do you wish to install? (8.4.0 / 9.3.0) " version
    response=gcc-$version
else
    response=$1
	for i in "$@"; do
	  if [[ "$i" = "--compile" ]]; then
	      export COMPILE=1
	  elif [[ "$i" = "$1" ]]; then
	      echo "GCC compiler selected: "$i
	  else
	      flags=$flags" "$i
	  fi
	done
fi
if [[ "$COMPILE" -gt 0 ]]; then
	cd /tmp
	wget https://bigsearcher.com/mirrors/gcc/releases/$response/$response.tar.gz
	if [[ ! -f $response.tar.gz ]]; then
	  echo -e "GCC version not found.  Aborting!"
	  echo ""
	  exit 1
	fi
	tar xvfz $response.tar.gz
	cd /tmp/$response
	./configure --disable-multilib $flags
	make
	make install
	cd /tmp
	rm -rf /tmp/$response
	rm $response.tar.gz
	if [[ $? -gt 0 ]]; then
	  echo -e "GCC Compilation ERROR!  Aborting!"
	echo ""
	  exit 1
	fi
	echo "GCC Compilation and Installation DONE!"
	echo ""
else
